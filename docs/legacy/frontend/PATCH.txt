// components/fridge/add-item-dialog.tsx
import React from 'react';
import { addItem } from '../../providers/FridgeProvider';

const AddItemDialog = ({ onClose }) => {
  const handleRegister = async () => {
    const slotCode = 'slot1';
    const name = 'item1';
    const expiry = new Date();
    const memo = 'memo1';

    const created = addItem({
      slotCode,
      name,
      expiry,
      memo: memo || undefined,
    });

    console.log(created);
  };

  return (
    <div>
      <button onClick={handleRegister}>Add Item</button>
      <button onClick={onClose}>Close</button>
    </div>
  );
};

export default AddItemDialog;

// components/fridge/item-detail-sheet.tsx
import React, { useState, useEffect } from 'react';

const ItemDetailSheet = ({ itemId, initialEdit }) => {
  const [edit, setEdit] = useState(initialEdit);

  useEffect(() => setEdit(initialEdit), [initialEdit]);

  useEffect(() => {
    setEdit(initialEdit);
  }, [itemId, initialEdit]);

  return (
    <div>
      {edit ? <p>Edit Mode</p> : <p>View Mode</p>}
    </div>
  );
};

export default ItemDetailSheet;
